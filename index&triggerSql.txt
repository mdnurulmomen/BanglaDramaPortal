create or replace TRIGGER TRIGGER_TIME_DRAMACONTENT BEFORE UPDATE ON DRAMACONTENT
FOR EACH ROW 
BEGIN
    IF :NEW.TITLE != :OLD.TITLE OR :NEW.STATUS != :OLD.STATUS THEN
       :NEW.UPLOAD_DATE := sysdate;
    END IF;
END;
